create database student_att;
use student_att;
CREATE TABLE ADDEMP(EMPNAME VARCHAR(70),
FATHERNAME VARCHAR(70),
GENDER VARCHAR(12),
BIRTHDATE DATE,
QUALIFICATION VARCHAR(70),
DESIGNATION VARCHAR(70),
CONTACTNO VARCHAR(12),
MAIL VARCHAR(100),
ADDRESS VARCHAR(100),
USERID VARCHAR(70)PRIMARY KEY,
PASS VARCHAR(70));


CREATE TABLE ADDSTD(STDNAME VARCHAR(70),
FATHERNAME VARCHAR(70),
GENDER VARCHAR(12),
BIRTHDATE DATE,
HIGHERSTUDIES VARCHAR(70),
COURSE VARCHAR(70),
CONTACT VARCHAR(12),
MAIL VARCHAR(100),
ADDRESS VARCHAR(100),
TAGID VARCHAR(50)) ;


CREATE TABLE ADDCLASS(PROFESSORNAME VARCHAR(70),
CLASSNAME VARCHAR(70),
CLASSTIME TIME,
CLASSDATE DATE);

CREATE TABLE ATTENDENCE(PROFESSORNAME VARCHAR(70),
CLASSNAME VARCHAR(70),TAGID VARCHAR(30),
INTIME TIME,
OUTTIME TIME,
LECTUREDATE DATE) ;

CREATE TABLE TEMPCLASS(PNAME VARCHAR(70),CNAME VARCHAR(70)) ;
